import { marked } from 'marked'

marked.setOptions({
  breaks: true,
  gfm: true
})

export function renderMarkdown(content: string): string {
  const html = marked.parse(content) as string
  
  // Basic sanitization - same as app frontend
  return html
    .replace(/<script\b[^<]*(?:(?!<\/script>)<[^<]*)*<\/script>/gi, '')
    .replace(/on\w+="[^"]*"/gi, '')
    .replace(/on\w+='[^']*'/gi, '')
}
```

const formattedBody = computed(() => {
  const text = props.comment.displayBody || props.comment.body
  
  // Escape HTML to prevent XSS
  let escaped = text
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;')
  
  // Wrap @mentions in styled spans using actual mention names
  // Format: @Name (from displayBody field)
  if (props.comment.mentions && props.comment.mentions.length > 0) {
    for (const mention of props.comment.mentions) {
      const escapedName = mention.name
        .replace(/[.*+?^${}()|[\]\\]/g, '\\$&')
      const regex = new RegExp(`@${escapedName}(?!\\w)`, 'g')
      escaped = escaped.replace(regex, `<span class="mention">@${mention.name}</span>`)
    }
  }
  
  return linkifyText(escaped)
})
```
